#include <stdio.h>

int fib(int n) {
  if (n = 0) {
    return 0;
  }
  if (n == 1) {
    return 1;
  }
  return fib(n-1) + fib(n-1);
}

int main() {
  int n = 10;
  for (int i=0; i<=n; i++) {
    printf("%d ", fib(i));
  }
  return 0;
}

Identified issues:

  • Typo in base case (= instead of ==)
  • Incorrect recursive call (fib(n-1) + fib(n-1))
  • Performance can be improved

Here is the new code with highlighted changes: 

#include <stdio.h>

int fib(int n) {
  if (n == 0) { // fix
    return 0;
  }
  if (n == 1) {
    return 1;
  }
  return fib(n-1) + fib(n-2); // fix
}

int main() {
  int n = 10;
  for (int i=0; i<=n; i++) {
    printf("%d ", fib(i));
  }
  return 0;
}
